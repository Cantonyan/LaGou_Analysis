def Sort_Skill(df):
    Job_num = len(df)
    df_skill = df.iloc[:,0:13] #
    Label = list(df_skill.axes[1]) #
    SkillMat = np.array(df_skill) #
    SkillSum = SkillMat.sum(axis=0) #
    SkillNum_sort = sorted(SkillSum, reverse=True) #
    Index_sort = np.argsort(-SkillSum) #
    Labels_sort = [] #
    for i in range(13): 
            Label_sort = Label[Index_sort[i]]
            Labels_sort.append(Label_sort) #
    return Job_num, SkillNum_sort, Labels_sort

def Draw_Skill(Job_num, SkillNum_sort, Labels_sort, Title):
    plt.rcParams['font.sans-serif'] = ['simHei'] 
    plt.rcParams['axes.unicode_minus'] = False
    plt.figure(figsize=(10,5), dpi = 100) #
    ax1=plt.subplot(111) #
    width=0.5 #
    x_bar=np.arange(13)
    rect=ax1.bar(x=x_bar,height=SkillNum_sort,width=width,color="lightblue") # 
    for rec in rect: #
        x=rec.get_x()
        height=rec.get_height()
        ax1.text(x+0.15,1.02*height,str(int(height)))

    ax1.set_xticks(x_bar) 
    ax1.set_xticklabels(Labels_sort) #X
    ax1.set_ylabel("(%d" %(Job_num)) #y
    ax1.set_title(Title) #
    plt.show()
